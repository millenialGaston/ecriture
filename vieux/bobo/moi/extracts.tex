\begin{verse}

	\poemtitle{A nice girl named Valeria/Who has a delicious posterior}

I love a tall girl. When we lie in\\
   Bed\\
She on her back and I stretched\\
   Upon her,\\
And our middle parts are busy\\
   With each other,\\
My toes play with her toes and my\\
   Tongue with her tongue\\
And all the parts are happy\\
   Because she is a tall girl\\

\attrib{T.S. Eliot (1888--1965)}

\end{verse}
